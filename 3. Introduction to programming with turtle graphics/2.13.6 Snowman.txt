def snowman(l):
    setposition(0,(l*2+(l/2)*2+(l/4)*2)*-1)
    begin_fill()
    pendown()
    color("gray")
    circle(l)
    end_fill()
    setposition(0,(l/2*2+l/4*2)*-1)
    pendown()
    begin_fill()
    circle(l/2)
    end_fill()
    setposition(0,(l/4*2)*-1)
    begin_fill()
    circle(l/4)
    end_fill()
    setposition(0,l/4)
    penup()
speed(10)
tam = int(input("ingrese la longitud del tam: "))
snowman(tam)
